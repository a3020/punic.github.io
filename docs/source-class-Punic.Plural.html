 26      * Return the plural rule ('zero', 'one', 'two', 'few', 'many' or 'other') for a number and a locale
 27      * @param string|int|float $number The number to check the plural rule for for
 28      * @param string $locale='' The locale to use. If empty we'll use the default locale set in \Punic\Data
 29      * @return string Returns one of the following values: 'zero', 'one', 'two', 'few', 'many', 'other'
 30      * @throws \Punic\Exception\BadArgumentType Throws a \Punic\Exception\BadArgumentType if $number is not a valid number
 31      * @throws \Exception Throws a \Exception if there were problems calculating the plural rule
 32      */
 33     public static function getRule($number, $locale = '')
 34     {
 35         if (is_int($number)) {
 36             $intPartAbs = strval(abs($number));
 37             $floatPart = '';
 38         } elseif (is_float($number)) {
 39             $s = strval($number);
 40             if (strpos($s, '.') === false) {
 41                 $intPart = $s;
 42                 $floatPart = '';
 43             } else {
 44                 list($intPart, $floatPart) = explode('.', $s);
 45             }
 46             $intPartAbs = strval(abs(intval($intPart)));
 47         } elseif (is_string($number) && isset($number[0])) {
 48             if (preg_match('/^[+|\\-]?\\d+\\.?$/', $number)) {
 49                 $v = intval($number);
 50                 $intPartAbs = strval(abs($v));
 51                 $floatPart = '';
 52             } elseif (preg_match('/^(\\d*)\\.(\\d+)$/', $number, $m)) {
 53                 list($intPart, $floatPart) = explode('.', $number);
 54                 $v = @intval($intPart);
 55                 $intPartAbs = strval(abs($v));
 56             } else {
 57                 throw new Exception\BadArgumentType($number, 'number');
 58             }
 59         } else {
 60             throw new Exception\BadArgumentType($number, 'number');
 61         }
 62         // 'n' => '%1$s', // absolute value of the source number (integer and decimals).
 63         $v1 = $intPartAbs . (strlen($floatPart) ? ".$floatPart" : '');
 64         // 'i' => '%2$s', // integer digits of n
 65         $v2 = $intPartAbs;
 66         // 'v' => '%3$s', // number of visible fraction digits in n, with trailing zeros.
 67         $v3 = strlen($floatPart);
 68         // 'w' => '%4$s', // number of visible fraction digits in n, without trailing zeros.
 69         $v4 = strlen(rtrim($floatPart, '0'));
 70         // 'f' => '%5$s', // visible fractional digits in n, with trailing zeros.
 71         $v5 = strlen($floatPart) ? strval(intval($floatPart)) : '0';
 72         // 't' => '%6$s', // visible fractional digits in n, without trailing zeros.
 73         $v6 = trim($floatPart, '0');
 74         if (!isset($v6[0])) {
 75             $v6 = '0';
 76         }
 77         $result = 'other';
 78         $node = \Punic\Data::getLanguageNode(\Punic\Data::getGeneric('plurals'), $locale);
 79         foreach ($node as $rule => $formulaPattern) {
 80             $formula = sprintf($formulaPattern, $v1, $v2, $v3, $v4, $v5, $v6);
 81             $check = str_replace(array('static::inRange(', ' and ', ' or ', ', false, ', ', true, ', ', array('), ' , ', $formula);
 82             if (preg_match('/[a-z]/', $check)) {
 83                 throw new \Exception('Bad formula!');
 84             }
 85             // fix for difference in modulo (%) in the definition and the one implemented in PHP for decimal numbers
 86             while (preg_match('/(\\d+\\.\\d+) % (\\d+(\\.\\d+)?)/', $formula, $m)) {
 87                 list(, $decimalPart) = explode('.', $m[1], 2);
 88                 $decimals = strlen(rtrim($decimalPart, '0'));
 89                 if ($decimals > 0) {
 90                     $pow = intval(pow(10, $decimals));
 91                     $repl = '(' . strval(intval(floatval($m[1]) * $pow)) . ' % ' . strval(intval(floatval($m[2] * $pow))) . ') / ' . $pow;
 92                 } else {
 93                     $repl = strval(intval($m[1])) . ' % ' . $m[2];
 94                 }
 95                 $formula = str_replace($m[0], $repl, $formula);
 96             }
 97             $formulaResult = @eval("return ($formula) ? 'yes' : 'no';");
 98             if ($formulaResult === 'yes') {
 99                 $result = $rule;
100                 break;
101             } elseif ($formulaResult !== 'no') {
102                 throw new \Exception('There was a problem in the formula ' . $formulaPattern);
103             }
104         }
105 
106         return $result;
107     }

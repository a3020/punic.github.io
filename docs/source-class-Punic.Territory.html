 82     /**
 83      * Return a list of some specified territory, structured or not.
 84      * $levels control which data you want to retrieve. It can be one or more of the following values:
 85      * <ul>
 86      *     <li>'W': world</li>
 87      *     <li>'C': continents</li>
 88      *     <li>'S': sub-continents</li>
 89      *     <li>'c': countries</li>
 90      * </ul>
 91      * If only one level is specified you'll get a flat list (like the one returned by {@link getContinents}).
 92      * If one or more levels are specified, you'll get a structured list (like the one returned by {@link getContinentsAndCountries}).
 93      * @param string $levels A string with one or more of the characters: 'W' (for world), 'C' (for continents), 'S' (for sub-continents), 'c' (for countries)
 94      * @param string $locale = '' The locale to use. If empty we'll use the default locale set in \Punic\Data
 95      * @return array
 96      * @link http://www.unicode.org/cldr/charts/latest/supplemental/territory_containment_un_m_49.html
 97      */
 98     public static function getList($levels = 'W', $locale = '')
 99     {
100         static $levelMap = array('W' => 0, 'C' => 1, 'S' => 2, 'c' => 3);
101         $decodedLevels = array();
102         $n = is_string($levels) ? strlen($levels) : 0;
103         if ($n > 0) {
104             for ($i = 0; $i < $n; $i++) {
105                 $l = substr($levels, $i, 1);
106                 if (!array_key_exists($l, $levelMap)) {
107                     $decodedLevels = array();
108                     break;
109                 }
110                 if (!in_array($levelMap[$l], $decodedLevels, true)) {
111                     $decodedLevels[] = $levelMap[$l];
112                 }
113             }
114         }
115         if (count($decodedLevels) === 0) {
116             throw new \Punic\Exception\BadArgumentType($levels, "list of territory kinds: it should be a list of one or more of the codes '" . implode("', '", array_keys($levelMap)) . "'");
117         }
118         $struct = self::filterStructure(self::getStructure(), $decodedLevels, 0);
119         $flatList = (count($decodedLevels) > 1) ? false : true;
120         $finalized = self::finalizeWithNames(Data::get('territories', $locale), $struct, $flatList);
121 
122         if ($flatList) {
123             natcasesort($finalized);
124         } else {
125             $finalized = static::sort($finalized);
126         }
127 
128         return $finalized;
129     }

145     protected static function fillStructure($data, $id, $level)
146     {
147         $item = array('id' => $id, 'level' => $level, 'children' => array());
148         if (array_key_exists($id, $data)) {
149             foreach ($data[$id]['contains'] as $childID) {

157     protected static function finalizeWithNames($data, $list, $flatList)
158     {
159         $result = array();
160         foreach ($list as $item) {
161             $name = $data[$item['id']];
162             if ($flatList) {
163                 $result[$item['id']] = $name;
164             } else {
165                 $result[$item['id']] = array('name' => $name);
166                 if (count($item['children']) > 0) {
167                     $result[$item['id']]['children'] = static::finalizeWithNames($data, $item['children'], $flatList);
168                 }
169             }
170         }
171 
172         return $result;
173     }

175     protected static function filterStructure($parent, $levels)
176     {
177         $thisResult = array();
178         if (in_array($parent['level'], $levels, true)) {
179             $thisResult[0] = $parent;
180             $thisResult[0]['children'] = array();
181             $addToSub = true;
182         } else {
183             $addToSub = false;
184         }
185 
186         $subList = array();
187         foreach ($parent['children'] as $child) {
188             $subList = array_merge($subList, static::filterStructure($child, $levels));
189         }
190         if ($addToSub) {
191             $thisResult[0]['children'] = $subList;
192         } else {
193             $thisResult = $subList;
194         }
195 
196         return $thisResult;
197     }

199     protected static function sort($list)
200     {
201         foreach (array_keys($list) as $i) {
202             if (array_key_exists('children', $list[$i])) {
203                 $list[$i]['children'] = static::sort($list[$i]['children']);
204             }
205         }
206         uasort($list, function ($a, $b) {
207             return strcasecmp($a['name'], $b['name']);
208         });
209 
210         return $list;
211     }
